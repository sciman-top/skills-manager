    """Staleness detection for scheduled data sources.

    Thresholds per D-13: SODIR daily (36h buffer), BSEE weekly (10d buffer),
    EIA monthly (45d buffer). Staleness = last_success older than threshold.
    """
    import logging
    from datetime import datetime, timedelta
    from typing import Any, Dict, List

    logger = logging.getLogger(__name__)

    # Per D-13: cadence + buffer = threshold
    # SODIR: daily + 12h buffer = 36 hours
    # BSEE: weekly + 3 days buffer = 10 days
    # EIA: monthly + 15 days buffer = 45 days
    STALENESS_THRESHOLDS: Dict[str, timedelta] = {
        "sodir_refresh": timedelta(hours=36),
        "bsee_refresh": timedelta(days=10),
        "eia_us_refresh": timedelta(days=45),
    }

    def check_staleness(status: dict) -> List[str]:
        """Return list of job names that are stale.

        Args:
            status: Status dict from StatusReporter.build_report().
                    Expected shape: {"jobs": {name: {"start_time": iso_str, ...}}}

        Returns:
            List of stale job names.
        """
        stale = []
        now = datetime.now()
        jobs = status.get("jobs", {})
        for job_name, threshold in STALENESS_THRESHOLDS.items():
            job_status = jobs.get(job_name, {})
            last_run_str = job_status.get("start_time") or job_status.get("last_run")
            if last_run_str is None:
                stale.append(job_name)
                continue
            try:
                last_run = datetime.fromisoformat(last_run_str)
            except (ValueError, TypeError):
                stale.append(job_name)
                continue
            if (now - last_run) > threshold:
                stale.append(job_name)
        return stale

    def get_staleness_details(status: dict) -> List[Dict[str, Any]]:
        """Return staleness details for all tracked sources.

        Returns list of dicts with keys: job_name, hours_since_last_run,
        threshold_hours, is_stale.
        """
        details = []
        now = datetime.now()
        jobs = status.get("jobs", {})
        for job_name, threshold in STALENESS_THRESHOLDS.items():
            job_status = jobs.get(job_name, {})
            last_run_str = job_status.get("start_time") or job_status.get("last_run")
            if last_run_str is None:
                hours_since = None
                is_stale = True
            else:
                try:
                    last_run = datetime.fromisoformat(last_run_str)
                    hours_since = (now - last_run).total_seconds() / 3600
                    is_stale = (now - last_run) > threshold
                except (ValueError, TypeError):
                    hours_since = None
                    is_stale = True
            details.append({
                "job_name": job_name,
                "hours_since_last_run": hours_since,
                "threshold_hours": threshold.total_seconds() / 3600,
                "is_stale": is_stale,
            })
        return details
    ```

    """Email alerting for scheduler failures and staleness.

    Per D-15: SMTP configured via .env (SMTP_HOST, SMTP_USER, SMTP_PASS).
    Per D-16: Alert on (1) job failure after all retries, (2) staleness threshold breached.
    Partial success (0 records) does NOT trigger alerts.
    """
    import logging
    import smtplib
    from email.mime.text import MIMEText
    from email.mime.multipart import MIMEMultipart
    from typing import List, Optional

    logger = logging.getLogger(__name__)

    class AlertSender:
        def __init__(self, smtp_host: Optional[str], smtp_user: Optional[str],
                     smtp_pass: Optional[str], smtp_port: int = 587,
                     recipients: Optional[List[str]] = None):
            self.smtp_host = smtp_host
            self.smtp_user = smtp_user
            self.smtp_pass = smtp_pass
            self.smtp_port = smtp_port
            self.recipients = recipients or ([smtp_user] if smtp_user else [])
            self._enabled = bool(smtp_host and smtp_user and smtp_pass)

        @property
        def enabled(self) -> bool:
            return self._enabled

        def send_alert(self, subject: str, body: str) -> None:
            if not self._enabled:
                logger.warning("SMTP not configured; alert logged only: %s", subject)
                return
            try:
                msg = MIMEMultipart()
                msg["From"] = self.smtp_user
                msg["To"] = ", ".join(self.recipients)
                msg["Subject"] = subject
                msg.attach(MIMEText(body, "plain"))
                with smtplib.SMTP(self.smtp_host, self.smtp_port) as server:
                    server.starttls()
                    server.login(self.smtp_user, self.smtp_pass)
                    server.send_message(msg)
                logger.info("Alert sent: %s", subject)
            except Exception as exc:
                logger.error("Failed to send alert '%s': %s", subject, exc)
    ```
