```bash
# Output validation (D-12)
validate_output() {
    local file="$1"
    local missing=()
    grep -qi "key findings" "$file"        || missing+=("Key Findings")
    grep -qi "relevance" "$file"           || missing+=("Relevance")
    grep -qi "recommended actions" "$file" || missing+=("Recommended Actions")
    if [[ ${#missing[@]} -gt 0 ]]; then
        echo "${missing[*]}"
        return 1
    fi
    return 0
}
```
