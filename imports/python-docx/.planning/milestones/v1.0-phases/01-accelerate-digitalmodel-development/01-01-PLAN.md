    ```python
    from pydantic import BaseModel, Field, field_validator
    from pathlib import Path
    import yaml

    class StandardRef(BaseModel):
        id: str                    # e.g. "DNV-RP-F109"
        edition: int | str         # e.g. 2021 or "2021-09"
        title: str                 # Full title of the standard

    class FunctionEntry(BaseModel):
        name: str                  # Python function name
        clause: str                # e.g. "S3.2.1"
        equation: str = ""         # e.g. "Eq 3.1" (optional for lookup/table functions)
        description: str           # Human-readable description
        inputs: list[str] = []     # Parameter names
        outputs: list[str] = []    # Return value descriptions

    class ModuleManifest(BaseModel):
        module: str                # Module path e.g. "subsea/on_bottom_stability"
        description: str           # Module description
        primary_standard: StandardRef
        functions: list[FunctionEntry] = Field(..., min_length=1)

    def validate_manifest_file(path: Path | str) -> ModuleManifest:
        """Load and validate a manifest.yaml file. Raises ValueError with path context on failure."""
        path = Path(path)
        with open(path) as f:
            data = yaml.safe_load(f)
        try:
            return ModuleManifest(**data)
        except Exception as e:
            raise ValueError(f"Invalid manifest at {path}: {e}") from e
    ```

    Create `digitalmodel/scripts/validate_manifests.py` that:

    1. Recursively finds all `manifest.yaml` files under `src/digitalmodel/`
    2. Validates each using `validate_manifest_file()` from the schema module
    3. Prints summary: N manifests found, N valid, N invalid
    4. Exits 0 if all valid, exits 1 if any invalid (CI-friendly)
    5. Uses `argparse` with `--dir` flag defaulting to `src/digitalmodel/`

    Script structure:
    ```python
    #!/usr/bin/env python3
    """Validate all manifest.yaml files against the ModuleManifest schema.

    Usage: uv run python scripts/validate_manifests.py [--dir src/digitalmodel/]
    Exit code: 0 if all valid, 1 if any invalid or none found.
    """
    import argparse
    import sys
    from pathlib import Path

    from digitalmodel.specs.manifest_schema import validate_manifest_file

    def main():
        parser = argparse.ArgumentParser(description="Validate manifest.yaml files")
        parser.add_argument("--dir", default="src/digitalmodel/", help="Root directory to search")
        args = parser.parse_args()

        root = Path(args.dir)
        manifests = sorted(root.rglob("manifest.yaml"))

        if not manifests:
            print(f"No manifest.yaml files found under {root}")
            sys.exit(1)

        errors = []
        for mf in manifests:
            try:
                result = validate_manifest_file(mf)
                print(f"  OK  {mf} ({len(result.functions)} functions)")
            except ValueError as e:
                print(f"  FAIL {mf}: {e}")
                errors.append(mf)

        print(f"\nSummary: {len(manifests)} manifests, {len(manifests)-len(errors)} valid, {len(errors)} invalid")
        sys.exit(1 if errors else 0)

    if __name__ == "__main__":
        main()
    ```
