    ```python
    from pydantic import BaseModel, field_validator
    from pathlib import Path
    from typing import Optional
    import pandas as pd

    class RigidJumperSpec(BaseModel):
        component_id: str
        manufacturer: str
        od_in: float
        wall_thickness_in: float
        length_ft: float
        material_grade: str
        pressure_rating_psi: float
        weight_air_kips: Optional[float] = None
        connection_type: Optional[str] = None
        data_source: str
        notes: Optional[str] = None

        @field_validator("od_in")
        @classmethod
        def od_positive(cls, v):
            if v <= 0:
                raise ValueError("OD must be positive")
            return v

    def load_rigid_jumpers(csv_path: Path) -> list[RigidJumperSpec]:
        df = pd.read_csv(csv_path)
        return [RigidJumperSpec(**{k.lower(): v for k, v in row.items() if pd.notna(v)}) for _, row in df.iterrows()]
    ```

    `mooring.py`:
    ```python
    from pydantic import BaseModel, field_validator
    from pathlib import Path
    from typing import Optional
    import pandas as pd

    class MooringComponentSpec(BaseModel):
        component_id: str
        component_type: str  # chain, wire_rope, polyester_rope, anchor, connector
        manufacturer: str
        size_mm: float
        mbl_kn: float  # minimum breaking load in kN
        weight_kg_per_m: Optional[float] = None
        material: Optional[str] = None
        grade: Optional[str] = None
        data_source: str
        notes: Optional[str] = None

        @field_validator("component_type")
        @classmethod
        def valid_type(cls, v):
            valid = {"chain", "wire_rope", "polyester_rope", "anchor", "connector"}
            if v not in valid:
                raise ValueError(f"component_type must be one of {valid}")
            return v

    def load_mooring_components(csv_path: Path) -> list[MooringComponentSpec]:
        df = pd.read_csv(csv_path)
        return [MooringComponentSpec(**{k.lower(): v for k, v in row.items() if pd.notna(v)}) for _, row in df.iterrows()]
    ```
